from ultralytics import YOLO
import cv2

# Load YOLO model (pretrained)
model = YOLO("yolov8n.pt")  # or "yolov8s.pt" for small

# Load your image
image_path = "car.jpg"  # replace with your image path
img = cv2.imread(image_path)

# Perform detection
results = model(img)

# Show results
results.show()
results.save("output.jpg")  # saves image with boxes
